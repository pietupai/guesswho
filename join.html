function joinGame() { const playerName = document.getElementById('playerName').value.trim(); const roomCode = document.getElementById('roomCode').value.trim(); if (!playerName) { alert('Please enter your name.'); console.warn('Player name is missing.'); return; } // Estä uusi painallus ja piilota painike const joinButton = document.querySelector('button'); joinButton.disabled = true; joinButton.style.display = 'none'; fetch('https://guesswho-backend.onrender.com/join-room', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ roomCode, playerName }), }) .then(response => response.json()) .then(data => { if (data.success) { alert(`Welcome ${playerName}, you successfully joined room: ${roomCode}`); console.log(`Player ${playerName} joined room ${roomCode}.`); } else { alert(data.message || 'Failed to join the room.'); console.error(`Join failed for room ${roomCode}: ${data.message}`); // Näytä painike uudelleen, jos liittyminen epäonnistui joinButton.disabled = false; joinButton.style.display = 'block'; } }) .catch(error => { console.error('Error joining room:', error); // Näytä painike uudelleen virhetilanteessa joinButton.disabled = false; joinButton.style.display = 'block'; }); }
